1、画布元素与图形上下文

  • getContext('2d') 获取图形上下文，对其进行绘制处理；
<canvas id="canvas" width="1000" height="1000" style="width: 1000px;height: 1000px;background-color: #D0D0D0"></canvas>

<script>
    var c = document.getElementById('canvas');
    var ctx = c.getContext('2d');

</script>

2、颜色、样式和阴影

fillStyle 填充

  • 颜色填充；
    ctx.fillStyle = '#FF0000';
    ctx.rect(0, 0, 1000, 900);
    ctx.fill();
  • 线性渐变填充
    var gradient = ctx.createLinearGradient(0, 0, 1000, 1000);// (x1,y1,x2,y2)
    gradient.addColorStop(0, 'red');
    gradient.addColorStop(1, 'yellow');

    ctx.fillStyle = gradient;
  • 圆形渐变填充
    var gradient = ctx.createRadialGradient(100, 100, 30, 100, 100, 80);// (x1,y1,r1,x2,y2,r2)
    gradient.addColorStop(0, 'red');
    gradient.addColorStop(1, 'yellow');

    ctx.fillStyle = gradient;
  • 图像填充
    var img = document.getElementById('lamp');// 也可以 new Image();
    img.onload = function () {
        var pat = ctx.createPattern(img, 'repeat');
        ctx.fillStyle = pat;
    };

strokeStyle 描边

  • 与 fillStyle 类似，略；
    ctx.strokeStyle = '#FF0000';
    ctx.rect(10, 10, 100, 100);
    ctx.stroke();

shadow 阴影

    ctx.shadowBlur = 20;
    ctx.shadowColor = 'black';
    ctx.shadowOffsetX = 20;
    ctx.shadowOffsetY = 20;

3、线条样式

    ctx.lineWidth = 10;// 线宽
    ctx.lineCap = 'square';// 结束帽 默认 butt 平直；round 圆；square 正方形；
    ctx.lineJoin = 'miter';// 折角 默认 miter 尖角；bevel 斜角；round 圆角
    ctx.miterLimit = 3;// miter 时，尖角限制；

4、矩形

    ctx.rect(20, 20, 100, 100);// 矩形路径 (x, y, w, h)
    ctx.strokeRect(100, 100, 200, 200);// 描边矩形
    ctx.fillRect(300, 300, 300, 300);// 填充矩形
    ctx.clearRect(0, 0, 300, 300);// 擦除 矩形内的绘制

5、路径

直线

    ctx.beginPath();
    ctx.moveTo(100, 100);// 起点
    ctx.lineTo(200, 200);
    ctx.lineTo(100, 200);
    ctx.lineTo(200, 120);
    ctx.closePath();// 连接起点和终点

贝塞尔曲线

    ctx.beginPath();
    ctx.moveTo(100, 100);// 起点
    
    ctx.quadraticCurveTo(120, 200, 200, 100);// （cx1, cy1, x2, y2）
    ctx.bezierCurveTo(120, 200, 180, 200, 400, 100);// (cx1, cy1, cx2, cy2, x2, y2)

圆

    ctx.arc(200, 200, 100, 0, Math.PI/2, true);// (x, y, r, startAngle, endAngle, clockwise) true = 逆时针

切弧

    ctx.beginPath();
    ctx.moveTo(200, 200);// 起点
    ctx.arcTo(400, 200, 300, 300, 50);// 3 点作 2 条线，圆弧与 2 直线相切 (x1, y1, x2, y2, r)

绘制

    ctx.stroke();
    ctx.fill();

裁剪：先绘制闭合区域，裁剪之后，只能显示绘制在区域内的内容；

    ctx.clip();

判断：点是否在路径上

    ctx.rect(20,20,150,100);
    if (ctx.isPointInPath(20, 50)) {
        ctx.stroke();
    } else {
        ctx.fill();
    }

6、转换

scale 缩放；

    ctx.scale(2, 3);// （x, y） 倍数
    ctx.rect(20, 20, 100,100);
    ctx.stroke();
    
    // 对之后绘制产生影响，x,y,w,h 统一乘以倍数；

rotate 旋转；

    ctx.rotate(Math.PI/8);
    
    // 同上

translate 重定义原点

    ctx.translate(70,70);
    
    // 同上

transform 2d 矩阵变化，综合上述变化

transform(a, b, c, d, e, f)

x2 = ax1 + cy1 + e;
y2 = bx1 + dy1 + f;

// a、b 控制缩放
// c、d 控制旋转
// e、f 控制平移

7、文本

    ctx.font="32px Arial";
    ctx.textAlign="center";
    ctx.textBaseline = "top";
    
    ctx.fillText("Top", 100, 100);
    ctx.strokeText("Hello World", 300, 100);

    var w = ctx.measureText('Hello').width;// 计算文本宽度
    console.log(w);

8、图片

    var img = new Image();
    img.src = 'icon.png';
    var w = img.width;
    var h = img.height;

    img.onload = function () {
        ctx.drawImage(img, 0, 0, w/2, h/3, 100, 150, w/2, h/3);// （裁剪位置，裁剪大小，图片位置，图片大小）
    }

9、像素

    var imgData = ctx.createImageData(300,300);

    for (var i=0;i<imgData.data.length;i+=4) {
        imgData.data[i+0] = parseInt(Math.random()*255);// r

    ctx.putImageData(imgData, 100, 200);// (x, y)

10、合成

  • 透明度
    ctx.globalAlpha = 0.5;
  • 层级关系
    ctx.globalCompositeOperation="destination-over";
    
    
//    source-over   默认。在目标图像上显示源图像。
//    source-atop   在目标图像顶部显示源图像。源图像位于目标图像之外的部分是不可见的。
//    source-in 在目标图像中显示源图像。只有目标图像之内的源图像部分会显示，目标图像是透明的。
//    source-out    在目标图像之外显示源图像。只有目标图像之外的源图像部分会显示，目标图像是透明的。
//    destination-over  在源图像上显示目标图像。
//    destination-atop  在源图像顶部显示目标图像。目标图像位于源图像之外的部分是不可见的。
//    destination-in    在源图像中显示目标图像。只有源图像之内的目标图像部分会被显示，源图像是透明的。
//    destination-out   在源图像之外显示目标图像。只有源图像之外的目标图像部分会被显示，源图像是透明的。
//    lighter   显示源图像 + 目标图像。
//    copy  显示源图像。忽略目标图像。
//    xor   使用异或操作对源图像与目标图像进行组合。
