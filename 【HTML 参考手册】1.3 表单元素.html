1、form 元素

<form  name="test-form" method="post" action="test.php" target="_blank"
       autocomplete="on" enctype="application/x-www-form-urlencoded"
       accept-charset="utf-8" novalidate></form>

    <!--
    accept-charset  默认是文档字符集，可设置表单字符集
    action  提交表单路径
    autocomplete    自动填表，默认 on
    enctype 表单编码方式，post 可用，默认 url 编码
    method  提交方式，默认 get
    novalidate  提交是不验证输入
    target  提交表单打开位置
    -->

2、input

有很多 h5 新增的 type，支持并不是很好，用基本几个可以满足要求了；

text

    <input name="user-id" form="form-id" autocomplete="on" autofocus 
           placeholder="用户名" maxlength="10" size="10" required>
           
    <!--
    name    提交时的 key
    value   提交时的 value
    form    绑定的表单，可以绑定多个
    autocomplete    自动填写
    autofocus   自动焦点
    placeholder 提示文本
    maxlength   最大限制长度
    size    输入框文本长度
    required    必填
    
    readonly    只读
    disabled    不可用
    multiple    多个
    pattern     正则过滤
    -->

button

<input type="button" value="点我" onclick="msg()">

checkbox

<input type="checkbox" checked name="vehicle[]" value="Bike"> 我有一辆自行车<br>
<input type="checkbox" name="vehicle[]" value="Car"> 我有一辆小轿车<br>
<input type="checkbox" name="vehicle[]" value="Boat"> 我有一艘船<br>

file

<input type="file" accept="image/png" name="head-image">

hidden

<input type="hidden" name="country" value="Norway">

image

    <input type="image" alt="图片" height="100" width="100" formaction="url" formmethod="post" formtarget="url">

password

<input type="password" name="pwd">

radio

<input type="radio" name="gender" value="男"> 男<br>
<input type="radio" name="gender" value="女"> 女

reset

<input type="reset">

submit

<input type="submit" formaction="url" formmethod="post" formtarget="url">

color

选择你喜欢的颜色: <input type="color" name="favcolor"><br>

date

生日: <input type="date" name="bday">

datetime

生日 (日期和时间): <input type="datetime" name="bdaytime">

datetime-local

生日 (日期和时间):<input type="datetime-local" name="bdaytime">

email

E-mail: <input type="email" name="usremail">

month

生日 ( 月和年 ): <input type="month" name="bdaymonth">

number

数量 ( 1 到 5 之间): <input type="number" name="quantity" min="1" max="5">

range

<input type="range" min="1" max="9" step="2" value="3">

search

Search Google: <input type="search" name="googlesearch">

tel

电话号码: <input type="tel" name="usrtel">

time

选择时间: <input type="time" name="usr_time">

url

添加你的主页: <input type="url" name="homepage">

week

选择周: <input type="week" name="week_year">

3、textarea

    <textarea rows="10" cols="30" placeholder="意见反馈" autofocus form="form-id" maxlength="200" name="option" required wrap="soft"></textarea>
    
    <!--
    rows    行
    cols    列
    wrap    是否提交换行
    -->

4、button

    <button type="submit" autofocus form="form-id" formaction="url" formmethod="post" 
            formtarget="_bland" formenctype="application/x-www-form-urlencoded" 
            formnovalidate name="ok" value="123">提交</button>

5、select 选择

    <select autofocus form="form-id" multiple="multiple" name="select" required size="3">
        <option value="volvo" selected label="vol">Volvo</option>
        <option value="saab" label="saa">Saab</option>
        <option value="mercedes" label="mer">Mercedes</option>
        <option value="audi" label="aud">Audi</option>
    </select>
    
    <!--
    multiple    按住 command 多选
    siize   显示个数
    label   显示标题，不设置时，显示内容
    -->

6、optgroup 分组选择

    <select>
        <optgroup label="水果">
            <option value="1">橘子</option>
            <option value="2">苹果</option>
        </optgroup>
        <optgroup label="蔬菜">
            <option value="11">青菜</option>
            <option value="22">萝卜</option>
        </optgroup>
    </select>

7、label 表单元素标题

    <label for="name" form="form-id">用户名</label>
    <input id="name">

8、fieldset 分组

    <fieldset name="sex" form="form-id">
        <legend>单选</legend>
        <input type="radio" id="male" name="sex" value="male" />男
        <br />
        <input type="radio" id="female" name="sex" value="female" />女
    </fieldset>

9、datalist 模糊输入选择

    <input list="browsers" name="browser">
    <datalist id="browsers">
        <option value="Internet Explorer">
        <option value="Firefox">
        <option value="Chrome">
        <option value="Opera">
        <option value="Safari">
    </datalist>

10、output 计算

    0<input type="range" id="a" value="50">100
    +
    <input type="number" id="b" value="50">
    =
    <output name="x" for="a b"></output>
</form>
